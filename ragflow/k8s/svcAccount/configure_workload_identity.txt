# binding svc account to the workloadidentity

gcloud builds submit --tag us-central1-docker.pkg.dev/greenfielddemos/greenfieldrepo/ragpipeline:v1 .


gcloud iam service-accounts add-iam-policy-binding \
    "genai-svc-acct@greenfielddemos.iam.gserviceaccount.com" \
    --member "serviceAccount:greenfielddemos.svc.id.goog[default/k8s-service-account]" \
    --role "roles/iam.workloadIdentityUser" \
    --project greenfielddemos
